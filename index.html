// 🛠️ 老師請注意：請務必將下方連結替換為您在 Teachable Machine 產生的模型網址 const URL = "https://teachablemachine.withgoogle.com/models/您的模型代碼/"; let model, webcam, labelContainer, maxPredictions; let isSetup = false; async function init() { if (isSetup) return; document.getElementById('start-btn').style.display = 'none'; document.getElementById('loading-text').style.display = 'block'; const modelURL = URL + "model.json"; const metadataURL = URL + "metadata.json"; try { // 1. 載入 AI 模型 model = await tmImage.load(modelURL, metadataURL); maxPredictions = model.getTotalClasses(); // 2. 設定相機參數（相容電腦與手機雙系統） const width = 400; const height = 300; const flip = false; webcam = new tmImage.Webcam(width, height, flip); // 【優化點】：先嘗試呼叫後鏡頭，若失敗(如電腦環境)則自動切換為預設鏡頭 try { await webcam.setup({ facingMode: "environment" }); } catch (e) { console.log("找不到後置鏡頭，切換為預設/前置鏡頭測試"); await webcam.setup(); // 預設盲抓（電腦視訊鏡頭就能動了） } // 3. 啟動鏡頭 await webcam.play(); document.getElementById('loading-text').style.display = 'none'; const videoElement = webcam.canvas; videoElement.setAttribute("playsinline", true); videoElement.setAttribute("muted", true); videoElement.style.width = "100%"; videoElement.style.height = "100%"; document.getElementById("webcam-container").appendChild(videoElement); // 4. 初始化結果顯示欄位 labelContainer = document.getElementById("label-container"); labelContainer.innerHTML = ""; for (let i = 0; i < maxPredictions; i++) { const row = document.createElement("div"); row.className = "label-row"; row.innerHTML = `--
0%`; labelContainer.appendChild(row); } isSetup = true; window.requestAnimationFrame(loop); } catch (error) { alert("系統啟動失敗！\n1. 請確認網址是否已改為您 Teachable Machine 的有效模型連結。\n2. 請確認瀏覽器是否已允許相機權限。"); console.error(error); document.getElementById('start-btn').style.display = 'block'; document.getElementById('loading-text').style.display = 'none'; } } async function loop() { webcam.update(); await predict(); window.requestAnimationFrame(loop); } async function predict() { const prediction = await model.predict(webcam.canvas); for (let i = 0; i < maxPredictions; i++) { const className = prediction[i].className; const probability = (prediction[i].probability * 100).toFixed(0); const row = labelContainer.childNodes[i]; row.querySelector('.class-name').innerText = className; row.querySelector('#prob-' + i).innerText = probability + "%"; row.querySelector('#bar-' + i).style.width = probability + "%"; if (probability > 70) { row.querySelector('#bar-' + i).style.backgroundColor = "#48bb78"; } else { row.querySelector('#bar-' + i).style.backgroundColor = "#cbd5e0"; } } }
